# 1. Start from the official lightweight Bun image
FROM oven/bun:1.1-slim

# 2. Install Git (needed to clone the repository)
RUN apt-get update && apt-get install -y git && rm -rf /var/lib/apt/lists/*

# 3. Set the working directory inside the container
WORKDIR /app

# 4. Clone the Interstellar-Astro repository directly into the container
RUN git clone https://github.com/UseInterstellar/Interstellar-Astro.git .

# 5. Install all dependencies using Bun
RUN bun install

# 6. Build the Astro production assets
RUN bun run build

# 7. Expose port 8080 for web traffic
EXPOSE 8080

# 8. Define the environment variable so the backend knows which port to use
ENV PORT=8080

# 9. Start the proxy server backend
CMD ["bun", "start"]
